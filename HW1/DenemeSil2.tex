\begin{problem}
Solve the equation $ 2x + 5 = -2 $ for $x$. \ANS{$-7/2$}
\begin{solution}[]
\begin{ssol}
$ 2x + 5 = -2 \implies 2x = -7 \implies x = -7/2 $
\end{ssol}
\begin{lsol}
Solve the equation $ 2x + 5 = -2 $ for $x$. Indeed,
\begin{alignat*}{2}
    2x + 5 &= -2 &&\qquad\text{given}\\
        2x &= -7 &&\qquad\text{subtract $5$ both sides}\\
         x &= \boxed{-\frac{7}{2}}
            &&\qquad\text{divide by $3$ both sides}\\
\end{alignat*}
\end{lsol}
\end{solution}
\end{problem}
